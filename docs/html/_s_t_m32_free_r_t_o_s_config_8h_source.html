190 /* The lowest interrupt priority that can be used in a call to a "set priority"

192 #define configLIBRARY_LOWEST_INTERRUPT_PRIORITY 0xf

194 /* The highest interrupt priority that can be used by any interrupt service
195 routine that makes calls to interrupt safe FreeRTOS API functions. DO NOT CALL
196 INTERRUPT SAFE FREERTOS API FUNCTIONS FROM ANY INTERRUPT THAT HAS A HIGHER
197 PRIORITY THAN THIS! (higher priorities are lower numeric values. */
198 #define configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY 5
199
200 /* Interrupt priorities used by the kernel port layer itself. These are generic
201 to all Cortex-M ports, and do not rely on any particular library functions. */
202 /* Warning in case of Ethernet, this prio (used by systick) should be higher (lower value) than ethernet Timer */
203 #define configKERNEL_INTERRUPT_PRIORITY ( 14 << (8 - configPRIO_BITS) )
