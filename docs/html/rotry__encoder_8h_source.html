67 encoder_data_t get_encoder_data(uint16_t pulses_per_rev, float check_dt){
68 uint32_t current_time = micros();
69 encoder_data_t encoder_data;
70
71 // Check if encoder has stopped
72 if(pulses_count > 0){
73 if (current_time - last_pulse_time > ENCODER_TIMEOUT_US) {
74 encoder_data.angV_rad_s = 0;
75 encoder_data.ang_rad = pulses_count*2.0*M_PI* last_direction;
76 return encoder_data; // No recent pulses, assume speed = 0
77 }
78
79 uint32_t pulse_dt_us = last_pulse_time - prev_pulse_time;
80 if (pulse_dt_us < ENCODER_MIN_DT_US) pulse_dt_us = ENCODER_MIN_DT_US; // Avoid division by zero
81
82 if (pulses_count <= MIN_PULSES_LS){
83 // Time between last two pulses (low speed method)
84 // If we only received very few pulses recently work with the last time difference
85 encoder_data.angV_rad_s = (2.0 * M_PI * last_direction) / ((pulse_dt_us / 1000000.0) * pulses_per_rev);
86 }else{
87 // High-speed method: Use pulse counting over fixed time
88 encoder_data.angV_rad_s = (pulses_count * 2.0 * M_PI * last_direction) / (pulses_per_rev * check_dt);
89 }
90 // Calculate angle
91 encoder_data.ang_rad = pulses_count*2.0*M_PI* last_direction;
92 // Reset pulse counter
93 encoder_data.pulses = pulses_count;
94 pulses_count = 0;
95 return encoder_data;
96 }else{
97 encoder_data.angV_rad_s = 0;
98 encoder_data.ang_rad = 0;
99 encoder_data.pulses = pulses_count;
100 return encoder_data;
101 }
102 }
